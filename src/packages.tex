% Start multi-subfigure environment: \figstart{main caption}
\newcommand{\figstart}[1]{%
  \begin{figure}[H]
    \centering
    \def\maincaption{#1}%
}

% End multi-subfigure environment: \figend
\newcommand{\figend}{%
  \ifx\maincaption\empty%
    % Do nothing - no caption at all
  \else%
    \caption*{\maincaption}%
  \fi%
  \end{figure}%
}
